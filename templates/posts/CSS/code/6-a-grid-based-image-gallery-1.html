      <script>
        function changeImage(imageBoxId) {
          const imageIdFront = imageBoxId + '-front';
          const imageIdBack = imageBoxId + '-back';
          const imageFront = document.querySelector(imageIdFront);
          const imageBack = document.querySelector(imageIdBack);
          imageFront.classList.toggle('screen_reader_only');
          imageBack.classList.toggle('screen_reader_only');
        }
  
        const numAlbums = 9;
        let i;
        for (i = 1; i < numAlbums + 1; i++) {
          let idName = '#image' + String(i);
          let album = document.querySelector(idName);
          album.addEventListener('click', function () {
            changeImage(idName);
          });
        }
      </script>
