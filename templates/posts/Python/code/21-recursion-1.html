def fib(n):
  """Assumes n int &gt;= 0
        Returns Fibonacci of n"""
  if n == 0:
      return 0
  elif n == 1 or n == 2:
      return 1
  else:
      return fib(n-1) + fib(n-2)


def fib_sequence(n):
  """Assumes n int &gt;= 0
    Returns sequence of Fibonacci numbers
    up to fib(n)"""

  for i in range(n+1):
      print(f'fib of {i} = {fib(i)}')


fib_sequence(20)  # outputs sequence of first 20 Fib numbers

print(fib(0))  # outputs 0
print(fib(20))  # outputs 6765
